Phase 5 — Jenkins

  • In Deploy stage: inject kubeconfig, run kubectl set image to update deployment
  • In Verify stage: kubectl rollout status deployment/eventapp — fails pipeline if rollout doesn't complete

  • Jenkins inside Docker needs kubectl installed in the container, or use a Jenkins agent with kubectl. Install kubectl in the Jenkins container image or via an init script.

deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: eventapp
spec:
  replicas: 2
  selector:
    matchLabels:
      app: eventapp
  strategy:
    type: RollingUpdate
    rollingUpdate:
      maxSurge: 1
      maxUnavailable: 0        # zero downtime
  template:
    metadata:
      labels:
        app: eventapp
    spec:
      containers:
      - name: eventapp
        image: yourusername/eventapp:latest
        ports:
        - containerPort: 8000
        env:
        - name: APP_VERSION
          value: "v1"
        - name: BUILD_NUMBER
          value: "1"
        - name: MONGO_URL
          value: "mongodb://mongo-service:27017"
        resources:
          requests:
            cpu: "100m"          # HPA baseline — required
            memory: "128Mi"
          limits:
            cpu: "500m"
            memory: "256Mi"
        livenessProbe:
          httpGet:
            path: /health
            port: 8000
          initialDelaySeconds: 10
          periodSeconds: 5
        readinessProbe:
          httpGet:
            path: /ready
            port: 8000
          initialDelaySeconds: 15
          periodSeconds: 5

service.yaml
apiVersion: v1
kind: Service
metadata:
  name: eventapp-service
spec:
  type: NodePort
  selector:
    app: eventapp
  ports:
  - port: 80
    targetPort: 8000
    nodePort: 30080
pipeline {
  agent any

  environment {
    IMAGE_NAME = "yourusername/eventapp"
    IMAGE_TAG  = "${BUILD_NUMBER}"
  }

  stages {

    stage('Clone') {
      steps {
        git branch: 'main', url: 'https://github.com/yourrepo/eventapp.git'
      }
    }

    stage('Install + Test') {
      steps {
        sh 'pip install -r requirements.txt'
        sh 'pytest tests/ -v --tb=short'
      }
    }

    stage('Build Image') {
      steps {
        sh "docker build -t ${IMAGE_NAME}:${IMAGE_TAG} ."
        sh "docker tag ${IMAGE_NAME}:${IMAGE_TAG} ${IMAGE_NAME}:latest"
      }
    }

    stage('Push to Docker Hub') {
      steps {
        withCredentials([usernamePassword(
          credentialsId: 'dockerhub',
          usernameVariable: 'DOCKER_USER',
          passwordVariable: 'DOCKER_PASS'
        )]) {
          sh 'echo $DOCKER_PASS | docker login -u $DOCKER_USER --password-stdin'
          sh "docker push ${IMAGE_NAME}:${IMAGE_TAG}"
          sh "docker push ${IMAGE_NAME}:latest"
        }
      }
    }

    stage('Deploy to Kubernetes') {
      steps {
        withCredentials([file(credentialsId: 'kubeconfig', variable: 'KUBECONFIG')]) {
          sh """
            kubectl set image deployment/eventapp \
              eventapp=${IMAGE_NAME}:${IMAGE_TAG} \
              --record
            kubectl set env deployment/eventapp \
              BUILD_NUMBER=${IMAGE_TAG}
          """
        }
      }
    }

    stage('Verify Rollout') {
      steps {
        withCredentials([file(credentialsId: 'kubeconfig', variable: 'KUBECONFIG')]) {
          sh 'kubectl rollout status deployment/eventapp --timeout=120s'
        }
      }
    }

  }

  post {
    failure {
      echo 'Pipeline failed — deployment not updated'
    }
    success {
      echo "Deployed ${IMAGE_NAME}:${IMAGE_TAG} successfully"
    }
  }
}
